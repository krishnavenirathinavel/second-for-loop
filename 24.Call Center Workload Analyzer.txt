 #include <stdio.h>

int main() {
    int N, calls;
    int total = 0;
    int overloaded = 0;
    scanf("%d", &N);
    for (int i = 0; i < N; i++) {
      scanf("%d", &calls);
      total += calls;         
      if (calls > 40) {   
        overloaded++;
        }
    }

    printf("Total Calls: %d\n", total);
    printf("Overloaded Hours: %d\n", overloaded);

    return 0;
}
